\begin{frame}

  \vspace{-0.6em}
  \begin{block}{科学问题}
    果蝇左右脑蘑菇体是否只是镜像结构，还是存在可传播到记忆环路的递质侧化？本项目用 FlyWire 连接组和公开可审计的赛博果蝇代理模型，将“右侧 5-HT 富集、左侧 Glu 偏置”的结构发现推进到 DPM 光遗传和 OCT/MCH 行为实验预测。
  \end{block}
  \refs{FlyWire: Dorkenwald et al., Nature 2024; Schlegel et al., Nature 2024. Mushroom body learning circuit: Aso et al., eLife 2014; Modi et al., Annu. Rev. Neurosci. 2020.}
\end{frame}

\begin{frame}{汇报结构与结论层级}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.50\textwidth}
    \begin{block}{四个问题}
      \begin{enumerate}
        \item KC 输入是否存在稳定递质侧化？
        \item 侧化 seed 是否能传播到记忆相关 readout？
        \item DPM 光遗传能否给出可测 release pattern？
        \item OCT/MCH 行为任务是否能承接这些预测？
      \end{enumerate}
    \end{block}
  \end{column}
  \begin{column}{0.46\textwidth}
    \begin{block}{当前结论}
      \begin{itemize}
        \item \orange{结构发现强}：5-HT 右偏，Glu 左偏，$\alpha'\beta'$ 最强。
        \item \orange{传播结果支持功能相关性}：readout 到 DAN、APL、DPM、MBON/MBIN。
        \item \orange{行为代理可用但有限}：OCT/MCH 方向稳定，MB 扰动未显著。
      \end{itemize}
    \end{block}
  \end{column}
\end{columns}
\end{frame}

\begin{frame}{实验 1：结构层递质侧化统计}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.48\textwidth}
    \begin{block}{数据与对象}
      \begin{itemize}
        \item 数据：FlyWire v783 neurons、synapses、cell annotations。
        \item 递质：per-synapse neurotransmitter prediction。
        \item 对象：左右半脑 Kenyon cells（KC），按 KC subtype 分组。
      \end{itemize}
    \end{block}
  \end{column}
  \begin{column}{0.48\textwidth}
    \begin{block}{分析和指标}
      \begin{itemize}
        \item 对每个 KC 计算上游输入递质比例。
        \item 左右半脑比较：Cohen's $d$，正值代表右偏，负值代表左偏。
        \item 统计控制：Mann--Whitney、permutation、bootstrap、FDR。
        \item 术语：5-HT 是血清素；Glu 是谷氨酸；$\alpha'\beta'$ 是记忆巩固相关 KC lobe。
      \end{itemize}
    \end{block}
  \end{column}
\end{columns}
\refs{Connectome and annotation basis: Dorkenwald et al., Nature 2024; Schlegel et al., Nature 2024. NT prediction: Eckstein et al., Cell 2024.}
\end{frame}

\begin{frame}{结果 1：KC 输入中 5-HT 右偏、Glu 左偏}
\scriptsize
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.58\textwidth}
    \includegraphics[width=0.96\textwidth]{Fig_ext1_cohens_d_heatmap_all_NT.png}
  \end{column}
  \begin{column}{0.38\textwidth}
    \begin{block}{主要观察}
      \begin{itemize}
        \item 5-HT/Serotonin：11/11 KC 类型右偏。
        \item Glu/Glutamate：7/11 KC 类型左偏且显著。
        \item 最强效应集中在 $\alpha'\beta'$。
        \item KC 输出仍以 ACh 为主，说明差异来自输入侧递质组成。
      \end{itemize}
    \end{block}
    \begin{block}{结论}
      左右蘑菇体不是简单镜像；至少在 KC 输入的化学组成上存在稳定侧化。
    \end{block}
  \end{column}
\end{columns}
\refs{Mushroom body architecture: Aso et al., eLife 2014; adult MB connectome: Li et al., eLife 2020; memory consolidation relevance: Krashes et al., Neuron 2007.}
\end{frame}

\begin{frame}{实验 2：连接组传播检验功能可达性}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.46\textwidth}
    \begin{block}{实验设计}
      \begin{itemize}
        \item 把连接组看成一个有向加权图：节点是神经元，边是突触连接强度。
        \item Seed 1：右侧 5-HT-rich KC，代表“右侧血清素富集输入”。
        \item Seed 2：左侧 Glu-rich KC，代表“左侧谷氨酸富集输入”。
        \item 对照：同侧、同数量随机 KC，判断结果是不是随便选一批 KC 都会出现。
        \item 传播：最多走 3 个突触层级，近似“上游差异能否传到下游记忆环路”。
      \end{itemize}
    \end{block}
  \end{column}
  \begin{column}{0.50\textwidth}
    \begin{block}{为什么这个实验有用}
      \begin{itemize}
        \item 如果 seed 只影响 KC 自己，说明结构侧化可能只是局部统计现象。
        \item 如果 seed 能传播到 DAN、APL、DPM、MBON/MBIN，说明它可能影响奖励、惩罚、记忆维持和行为输出。
        \item 这里验证的是\orange{功能可达性}，不是证明真实行为已经发生。
      \end{itemize}
    \end{block}
  \end{column}
\end{columns}
\refs{Graph-based connectome analysis is consistent with large-scale connectomics practice: Scheffer et al., eLife 2020; Dorkenwald et al., Nature 2024. MB circuit readouts: Aso et al., eLife 2014; Li et al., eLife 2020.}
\end{frame}

\begin{frame}{实验 2 指标解读：给计算机背景老师的读法}
\small
\begin{tabularx}{\textwidth}{@{}p{3.0cm}p{4.2cm}X@{}}
\toprule
\textbf{指标} & \textbf{计算含义} & \textbf{生物解释和用途} \\
\midrule
memory axis mass & 传播信号落到预定义记忆相关神经元集合的总量 & 类似图神经网络里某个子模块的 activation mass；越大表示侧化输入越容易影响记忆轴。\\
DAN 等 abs mass & 到达 DAN、APL、DPM、MBON 等神经元类别的信号强度，取绝对值，不区分正负 & DAN 像“教学信号”，APL/DPM 像“反馈和记忆维持”，MBON/MBIN 像“读出/调制接口”；用于判断影响到哪个功能模块。\\
response laterality & $(R-L)/(R+L)$，正值偏右，负值偏左 & 判断传播后的影响是否仍保留左右脑偏向，而不只是整体变强。\\
empirical $p$ & 和随机 seed 形成的 null distribution 比较 & 问题是“这个结果比随机挑 KC 更特殊吗”。\\
FDR $q$ & 多个指标一起检验后的假阳性校正 & 避免在很多脑区/指标里挑一个偶然显著的结果。\\
\bottomrule
\end{tabularx}
\vspace{0.4em}
\begin{block}{一句话}
这页的所有指标都在回答同一个问题：\blue{左右递质侧化是否能通过真实连接组传到记忆和行为相关模块，并且比随机对照更强。}
\end{block}
\end{frame}

\scriptsize
\begin{columns}[T,onlytextwidth]

    \includegraphics[width=\textwidth]{Fig2_functional_metric_heatmap.png}

      \begin{itemize}

\begin{frame}{实验 3：MB readout 到 DN/motor bridge}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.48\textwidth}
    \begin{block}{为什么做这个实验}
      只证明 KC 到 MB readout 还不够。行为必须经过下行神经元（descending neurons, DN）或其它运动出口。因此我们检查 MBON/DAN/APL/DPM 是否能在连接组传播中到达 DN family。
    \end{block}
  \end{column}
  \begin{column}{0.48\textwidth}
    \begin{block}{方法}
      \begin{itemize}
        \item Seed：left/right/bilateral MBON、DAN、APL/DPM、memory axis。
        \item 传播：FlyWire signed propagation。
        \item 输出：DN 数量、absolute mass、signed mass、laterality index。
        \item motor primitive 是透明 heuristic，不是 Eon 私有 DN-to-body 权重。
      \end{itemize}
    \end{block}
  \end{column}
\end{columns}
\end{frame}

\begin{frame}{实验 4：DPM 光遗传仿真验证}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.48\textwidth}
    \begin{block}{实验动机}
      DPM 是投射到蘑菇体的调制神经元，和嗅觉记忆稳定性相关。若结构侧化有功能意义，DPM 激活后应产生可测的左右 5-HT/KC readout 差异。
    \end{block}
    \begin{block}{湿实验可做性}
      DPM-driver 表达 CsChrimson 或 ReaChR；KC 或 MB compartment 表达 5-HT sensor / GCaMP。
    \end{block}
  \end{column}
  \begin{column}{0.48\textwidth}
    \begin{block}{仿真设置}
      \begin{itemize}
        \item Seed：left/right DPM。
        \item Opsin：CsChrimson、ReaChR、ChR2。
        \item 参数：波长、频率、脉宽、时长、光强。
        \item 控制：180 度旋转后按脑侧注册。
      \end{itemize}
    \end{block}
  \end{column}
\end{columns}
\refs{DPM and memory: Yu et al., Cell 2005; Keene and Waddell, Nat. Rev. Neurosci. 2007. CsChrimson: Klapoetke et al., Nat. Methods 2014. ReaChR: Lin et al., Nat. Neurosci. 2013.}
\end{frame}

\begin{frame}{实验 5：OCT/MCH 群体行为代理}

        \item 多重比较：FDR。
      \end{itemize}
    \end{block}
  \end{column}
\end{columns}

\begin{frame}{湿实验验证路径}
\small
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.32\textwidth}
    \begin{block}{结构证明}
      GRASP / split-GFP 验证右侧 DPM/5-HT 到右侧 $\alpha'\beta'$ KC，以及左侧 Glu 到左侧 $\alpha'\beta'$ KC。
    \end{block}
  \end{column}
  \begin{column}{0.32\textwidth}
    \begin{block}{功能证明}
      DPM-driver $>$ CsChrimson/ReaChR；KC 或 MB compartment 表达 5-HT sensor / GCaMP；做 180 度旋转控制。
    \end{block}
  \end{column}
  \begin{column}{0.32\textwidth}
    \begin{block}{行为证明}

\begin{frame}{核心结论}
\small
\begin{block}{发现}
果蝇蘑菇体 KC 输入存在稳定递质侧化：右侧更偏 5-HT，左侧更偏 Glu，最强在与记忆巩固相关的 $\alpha'\beta'$ 区。
\end{block}
\begin{block}{方法贡献}
公开可审计的赛博果蝇代理模型将该结构发现推进到连接组传播、DPM 光遗传预测和 OCT/MCH 行为实验设计。
\end{block}
\begin{alertblock}{边界}
当前还不能说连接组单独涌现完整行为，也不能说 MB 侧化已经被真实行为因果证明。下一步需要 DPM 成像、GRASP/split-GFP 和群体行为完成闭环。
\end{alertblock}
